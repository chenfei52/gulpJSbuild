获取收银台首页信息
/gateway/getPayMethod
POST JSON

请求参数
{
	"orderNum":"订单号"
}

成功返回参数

{
	"code": 0,  // 成功返回0，其他为失败
	"msg": "Success",  // 成功返回Success，失败返回失败原因
	"data": {
		"payMoney": 20000,  // 金额
		"bankMethod": ["BT", "M1", "I1"],  // 银行可用渠道 首页只显示可用渠道 对应的值的描述看最下面
		"walletMethod": ["OV"],  // 电子钱包可用渠道
		"storeMethod": ["FT"],  // 便利店可用渠道
		"onMethod": "M1",  // 已经选中的渠道，如果这个值不为空，跳转到对应的银行的详情页
		"expiryPeriod": 86384,  // 倒计时 单位：秒
		"customerName": "Oliver"  // 姓名
	}
}

失败返回参数

{
	"code": 1006, // 除了 0 其他都是失败
	"msg": "Data error",  // 失败原因
	"data": null
}
